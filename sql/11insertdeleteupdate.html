将查询结果插入记录
	insert into 表名1(属性列表1) select 属性列表2 from 表名2 where条件表达式。
	表名1：要插入的表。
	属性列表1:为哪些字段赋值
	属性列表2:从表中查出哪些字段的数据。
	属性列表1和属性列表2字段个数，及每个对应字段的数据类型是一样的。

create table medicine(id int,name varchar(20),function varchar(20),company varchar(20), address varchar(20));
create table product(id int,name varchar(20),function varchar(20),company varchar(20), address varchar(20));
insert into product(id,function,name,address,company) select  id,function,name,address,company from medicine;//把查询出来的多条记录都插入进去。
